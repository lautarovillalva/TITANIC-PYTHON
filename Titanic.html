x = df["Pclass"].sort_values(ascending=False).unique().astype('str')

# Bar plot

for index, value in enumerate(y):
    plt.text(value, index,
             str(value))
ax.set_ylabel('Clase de pasajero')
ax.set_title('Cantidad de sobrevivientes según la clase')
